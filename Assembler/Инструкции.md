### - Основные инструкции  
NOP // Пустая операция  
CMP AX, BX // Сравнить значения в регистрах AX и BX  
JMP label // Безусловный переход к метке (адресу) label  
JNZ/JNE label // Переход к метке label, если флаг ZF=1  
NEG AL // Знаковое отрицание значения в регистре AL. Эквивалент: AL = -AL  
  
### - Битовые операции  
AND AX, BX // Побитовое И. Эквивалент: AX = AX & BX  
OR AL, 10 // Побитовое ИЛИ. Эквивалент: AL = AL | 0x0A  
XOR AL, 8 // Побитовое исключающее ИЛИ. Эквивалент: AL = AL ^ 0x08  
NOT AL // Побитовое НЕ (инверсия всех битов)  
SHL AL,1 // Логический сдвиг влево  
SHR AL,1 // Логический сдвиг вправо  
SAR AL,1 // Арифметический сдвиг вправо (сохраняет знак)  
  
### - Арифметические  
INC AL // Увеличить значение в регистре AL на 1. Эквивалент: AL++  
DEC AL // Уменьшить значение в регистре AL на 1. Эквивалент: AL--  
ADD AL, BL // Сложить значения в регистрах AL и BL. Эквивалент: AL = AL + BL  
SUB AL, 6 // Вычесть 6 из значения в регистре AL. Эквивалент: AL = AL // 6  
IMUL rdi, rdi, 10 // Умножение со знаком. Эквивалент: rdi = rdi * 10  
MUL rdi // Беззнаковое умножение. Эквивалент: RAX = RAX * rdi  
DIV rdi // Беззнаковое деление. Эквивалент: RAX = RAX / rdi  
  
### - Перемещение данных  
MOV AL, 0 // Загрузить значение 0 в регистр AL. Эквивалент: AL = 0  
MOVZX AX,BL // Нулевое расширение. Эквивалент: AX = BL с нулями  
MOVSX AX,BL // Знаковое расширение. Эквивалент: AX = BL с копированием знака  
PUSH AX // Положить значение из регистра AX в стек (RSP -= 2)  
POP CX // Извлечь значение из стека в регистр CX (RSP += 2)  
LEA RAX,[RBX+RCX] // Загрузка эффективного адреса  
  
### - Прочее  
CALL subroutine // Вызвать подпрограмму (переход с сохранением адреса возврата)  
RET // Вернуться из подпрограммы (извлечь адрес возврата и перейти туда)  
INT 3 // Вызов отладчика (0xCC)  

HLT // Остановка процессора
